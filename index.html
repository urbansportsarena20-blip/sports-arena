var SHEET_PLAYERS = "Players"; function doGet(e) { var action = e.parameter.action; var result; if (action === "getBookings") result = getBookings(); else if (action === "getPlayers") result = getPlayers(); else if (action === "deleteBooking") result = deleteBooking(e.parameter.id); else if (action === "deletePlayer") result = deletePlayer(e.parameter.id); else result = { error: "Unknown action" }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action; var result; if (action === "addBooking") result = addBooking(data); else if (action === "editBooking") result = editBooking(data); else if (action === "addPlayer") result = addPlayer(data); else if (action === "editPlayer") result = editPlayer(data); else result = { error: "Unknown action" }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } // ── SHEET HELPER ───────────────────────────────────────────────── function getSheet(name) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === SHEET_BOOKINGS) { sheet.appendRow(["ID","Date","Day","Sport","Start","End","Customer","Phone","Total","Advance","Due","Status","CalendarEventId"]); sheet.getRange(1,1,1,13).setFontWeight("bold").setBackground("#1B5E20").setFontColor("white"); } else if (name === SHEET_PLAYERS) { sheet.appendRow(["ID","Name","Phone","Sport","Type"]); sheet.getRange(1,1,1,5).setFontWeight("bold").setBackground("#1565C0").setFontColor("white"); } } return sheet; } function generateId(prefix) { return prefix + Date.now().toString().slice(-6); } // ── BOOKINGS ───────────────────────────────────────────────────── function getBookings() { var sheet = getSheet(SHEET_BOOKINGS); var data = sheet.getDataRange().getValues(); var out = []; for (var i = 1; i < data.length; i++) { var r = data[i]; if (!r[0]) continue; out.push({ id: r[0], date: r[1] ? Utilities.formatDate(new Date(r[1]), "Asia/Kolkata", "yyyy-MM-dd") : "", day: r[2], sport: r[3], start: r[4] ? (r[4] instanceof Date ? Utilities.formatDate(r[4], "Asia/Kolkata", "HH:mm") : r[4].toString().substring(0,5)) : "", end: r[5] ? (r[5] instanceof Date ? Utilities.formatDate(r[5], "Asia/Kolkata", "HH:mm") : r[5].toString().substring(0,5)) : "", customer: r[6], phone: r[7], total: r[8], advance: r[9], due: r[10], status: r[11], calendarEventId: r[12] }); } return { success: true, data: out }; } function addBooking(data) { var sheet = getSheet(SHEET_BOOKINGS); var id = generateId("B"); var days = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday"]; var day = days[new Date(data.date).getDay()]; var due = (Number(data.total)||0) - (Number(data.advance)||0); sheet.appendRow([id, data.date, day, data.sport, data.start, data.end, data.customer, data.phone, data.total, data.advance, due, data.status, ""]); var calId = syncToCalendar(data.date, data.sport, data.start, data.end, data.customer, data.phone, data.total, data.status); if (calId) sheet.getRange(sheet.getLastRow(), 13).setValue(calId); return { success: true, id: id }; } function editBooking(data) { var sheet = getSheet(SHEET_BOOKINGS); var values = sheet.getDataRange().getValues(); var due = (Number(data.total)||0) - (Number(data.advance)||0); for (var i = 1; i < values.length; i++) { if (values[i][0] === data.id) { var row = i + 1; // Delete old calendar event var oldCalId = values[i][12]; if (oldCalId) { try { var ev = CalendarApp.getDefaultCalendar().getEventById(oldCalId); if (ev) ev.deleteEvent(); } catch(e) {} } var days = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday"]; var day = days[new Date(data.date).getDay()]; sheet.getRange(row, 1, 1, 12).setValues([[ data.id, data.date, day, data.sport, data.start, data.end, data.customer, data.phone, data.total, data.advance, due, data.status ]]); var newCalId = syncToCalendar(data.date, data.sport, data.start, data.end, data.customer, data.phone, data.total, data.status); if (newCalId) sheet.getRange(row, 13).setValue(newCalId); return { success: true }; } } return { error: "Booking not found" }; } function deleteBooking(id) { var sheet = getSheet(SHEET_BOOKINGS); var values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { if (values[i][0] === id) { var calId = values[i][12]; if (calId) { try { var ev = CalendarApp.getDefaultCalendar().getEventById(calId); if (ev) ev.deleteEvent(); } catch(e) {} } sheet.deleteRow(i + 1); return { success: true }; } } return { error: "Booking not found" }; } // ── CALENDAR SYNC ───────────────────────────────────────────────── function syncToCalendar(date, sport, start, end, customer, phone, total, status) { try { var sp = start.toString().split(":"); var ep = end.toString().split(":"); var sd = new Date(date); sd.setHours(parseInt(sp[0]), parseInt(sp[1]), 0); var ed = new Date(date); ed.setHours(parseInt(ep[0]), parseInt(ep[1]), 0); var event = CalendarApp.getDefaultCalendar().createEvent( sport + " — " + customer, sd, ed, { description: "Customer: "+customer+"\nPhone: "+phone+"\nTotal: Rs."+total+"\nStatus: "+status+"\nUrban Sports Arena, Jaipur" } ); return event.getId(); } catch(e) { return null; } } // ── PLAYERS ────────────────────────────────────────────────────── function getPlayers() { var sheet = getSheet(SHEET_PLAYERS); var data = sheet.getDataRange().getValues(); var out = []; for (var i = 1; i < data.length; i++) { if (!data[i][0]) continue; out.push({ id: data[i][0], name: data[i][1], phone: data[i][2], sport: data[i][3], type: data[i][4] }); } return { success: true, data: out }; } function addPlayer(data) { var sheet = getSheet(SHEET_PLAYERS); var id = generateId("P"); sheet.appendRow([id, data.name, data.phone, data.sport, data.type || "Walk-in"]); return { success: true, id: id }; } function editPlayer(data) { var sheet = getSheet(SHEET_PLAYERS); var values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { if (values[i][0] === data.id) { sheet.getRange(i+1, 1, 1, 5).setValues([[data.id, data.name, data.phone, data.sport, data.type]]); return { success: true }; } } return { error: "Player not found" }; } function deletePlayer(id) { var sheet = getSheet(SHEET_PLAYERS); var values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { if (values[i][0] === id) { sheet.deleteRow(i+1); return { success: true }; } } return { error: "Player not found" }; }
